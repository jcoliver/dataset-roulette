Bar plot

mean_female_per_category <- equal_pay %>% 
  group_by(major_category) %>% 
  summarize(mean = mean(percent_female))

mean_female_per_category
mean_female_per_category %>% 
  ggplot(aes(y = reorder(major_category, mean),
             x = mean,
             label = round(mean, 2))) +
  geom_col() +
  geom_label()
